#include"ns3/core-module.h"
#include"ns3/network-module.h"
#include"ns3/internet-module.h"
#include"ns3/point-to-point-module.h"
#include"ns3/applications-module.h"
#include<iostream>
#include<vector>
usingnamespace ns3;
// --- Server Application ---
classSingleServerApp:publicApplication{
public:
SingleServerApp():m_udpSocket(0),m_tcpSocket(0){}
private:
virtualvoidStartApplication(void){
// Setup UDP on Port 8080
m_udpSocket =Socket::CreateSocket(GetNode(),
UdpSocketFactory::GetTypeId());
m_udpSocket->Bind(InetSocketAddress(Ipv4Address::GetAny(),8080));
m_udpSocket->SetRecvCallback(MakeCallback
(&SingleServerApp::HandleReadUdp,this));
// Setup TCP on Port 8081

m_tcpSocket =Socket::CreateSocket(GetNode(),
TcpSocketFactory::GetTypeId());
m_tcpSocket->Bind(InetSocketAddress(Ipv4Address::GetAny(),8081));
m_tcpSocket->Listen();
m_tcpSocket->SetAcceptCallback(
MakeCallback(&SingleServerApp::HandleRequestTcp,this),
MakeCallback(&SingleServerApp::HandleAcceptTcp,this)
);
}
boolHandleRequestTcp(Ptr<Socket> socket,const Address& from){
returntrue;// Explicitly allow TCP connections
}
voidHandleAcceptTcp(Ptr<Socket> socket,const Address& from){
socket->SetRecvCallback(MakeCallback
(&SingleServerApp::HandleReadTcp,this));
m_tcpSockets.push_back(socket);// Store socket to prevent SIGSEGV
}
voidHandleReadUdp(Ptr<Socket> socket){
Ptr<Packet> packet; Address from;
while((packet = socket->RecvFrom(from))){
std::cout <<"[UDP] Server received "<< packet->GetSize()
<<" bytes at "<<Simulator::Now().GetSeconds()<<"
s\n";
}
}
voidHandleReadTcp(Ptr<Socket> socket){
Ptr<Packet> packet; Address from;
while((packet = socket->RecvFrom(from))){
if(packet->GetSize()>0){
std::cout <<"[TCP] Server received "<< packet->GetSize()
<<" bytes at "<<Simulator::Now().GetSeconds()
<<" s\n";
}
}
}
Ptr<Socket> m_udpSocket;
Ptr<Socket> m_tcpSocket;
std::vector<Ptr<Socket>> m_tcpSockets;
};
// --- Client Application ---

classSingleClientApp:publicApplication{
public:
SingleClientApp():m_socket(0){}
voidSetup(Address peer,uint32_t count, TypeId tid){
m_peer = peer; m_count = count; m_tid = tid;
}
private:
virtualvoidStartApplication(void){
m_socket =Socket::CreateSocket(GetNode(), m_tid);
m_socket->Connect(m_peer);
// Slight delay after connect to allow TCP handshake to complete
safely
Simulator::Schedule(Seconds(0.1),&SingleClientApp::SendPacket,
this);
}
voidSendPacket(){
if(m_count >0){
m_socket->Send(Create<Packet>(1024));// 1024 bytes packet size
m_count--;
Simulator::Schedule(Seconds(1.0),&SingleClientApp::SendPacket,
this);// 1 second interval
}
}
Ptr<Socket> m_socket;
Address m_peer;
uint32_t m_count;
TypeId m_tid;
};
