2.  Python/Scapy  I/O  Graph  Recreation  
The  five  graphs  were  recreated  using  Python  and  the  Scapy  library  to  match  the  Wireshark  UI  
visuals.
 
  import matplotlib.pyplot as plt  from scapy.all import rdpcap,  TCP,  UDP,  ARP,  IP   #  Load  the  pcap  file  pkts = rdpcap('24bce1580.pcap')  start_time = float(pkts[0].time)   def get_data(filter_func,  interval=0.5):      counts = {}      filtered = [p for p in pkts if filter_func(p)]      for p in filtered:          t = int((float(p.time)  - start_time)  / interval)          time_key = t * interval          counts[time_key]  = counts.get(time_key,  0)  + 1      return list(counts.keys()),  list(counts.values())   #  Filters  and  Colors  matched  to  your  Wireshark  UI  filters = [      ("All  Packets",  lambda p:  True,  "black"),      ("tcp",  lambda p:  p.haslayer(TCP),  "lawngreen"),      ("TLS",  lambda p:  p.haslayer(TCP)  and (p[TCP].dport  == 443 or p[TCP].sport  == 443),  "cyan"),      ("ARP",  lambda p:  p.haslayer(ARP),  "hotpink"),      ("UDP",  lambda p:  p.haslayer(UDP),  "saddlebrown")  ]   plt.figure(figsize=(12,  6))   for label,  func,  col in filters:      x,  y = get_data(func)      if x:  #  Only  plot  if  data  exists          lists = sorted(zip(x,  y))          x,  y = zip(*lists)

plt.plot(x,  y,  label=label,  color=col,  linewidth=1.2)   #  SETTING  THE  RANGE  TO  MATCH  YOUR  SCREENSHOT  plt.xlim(0,  85)    #  Time  in  seconds  plt.ylim(0,  80)    #  Packets  per  500ms  plt.yticks([0,  20,  40,  60,  80],  ['0  packets',  '20  packets',  '40  packets',  '60  packets',  '80  packets'])   plt.title("Wireshark  I/O  Graphs  Recreation")  plt.xlabel("Time  (s)")  plt.ylabel("Packets  /  500  ms")  plt.legend(loc='upper  right')  plt.grid(True,  linestyle=':',  alpha=0.6)   plt.savefig('io_graph_matched.png')  plt.show()    The  generated  image  io_graph_matched.png successfully  mirrors  the  trends  observed  in

3.  Protocol  Hierarchy  Metrics  
The  protocol  hierarchy  was  analyzed  to  extract  total  packet  counts  and  data  sizes.  
 import csv  from scapy.all import rdpcap   #  1.  Load  the  capture  file  pkts = rdpcap('24bce1580.pcap')   #  2.  Calculate  Metrics  total_packets = len(pkts)  total_captured_size = sum(len(p)  for p in pkts)  #  Payload  is  the  data;  Headers  =  Total  size  -  Payload  size  size_of_data = sum(len(p.payload)  for p in pkts)  size_of_headers = total_captured_size - size_of_data   #  3.  Print  results  to  console  as  required  print("---  Protocol  Hierarchy  Metrics  ---")  print(f"Total  number  of  packets:  {total_packets}")  print(f"Size  of  data:  {size_of_data} bytes")  print(f"Size  of  headers:  {size_of_headers} bytes")   #  4.  Save  to  CSV  with open('protocol_hierarchy_extracted.csv',  'w',  newline='')  as f:      writer = csv.writer(f)      writer.writerow(["Metric",  "Value"])      writer.writerow(["Total  Packets",  total_packets])      writer.writerow(["Size  of  Data  (Payload)",  size_of_data])      writer.writerow(["Size  of  Headers",  size_of_headers])   print("\nFile  saved:  protocol_hierarchy_extracted.csv")

4.  Conversation  Analysis  
The  capture  was  analyzed  for  unique  address  pairs  and  their  communication  metrics.  
  import csv  from scapy.all  import rdpcap,  IP,  TCP,  UDP  from collections import defaultdict   pkts = rdpcap('24bce1580.pcap')  stats = defaultdict(lambda:  {'packets':  0,  'bytes':  0,  'times':  [],  'protocols':  set()})   for p in pkts:      if IP  in p:          #  Identify  the  unique  pair  (A  <->  B)          pair = tuple(sorted([p[IP].src,  p[IP].dst]))                  #  Determine  protocol  for  this  packet          proto = "Other"          if p.haslayer(TCP):  proto = "TCP"          elif p.haslayer(UDP):  proto = "UDP"                  stats[pair]['packets']  += 1          stats[pair]['bytes']  += len(p)          stats[pair]['times'].append(float(p.time))          stats[pair]['protocols'].add(proto)   #  Find  unique  address  pair  with  maximum  bytes  max_pair = max(stats,  key=lambda k:  stats[k]['bytes'])

print(f"Max  Communication  Pair:  {max_pair} ({stats[max_pair]['bytes']} 
bytes)")   #  Export  to  CSV  and  calculate  Inter-packet  time  with open('conversations.csv',  'w',  newline='')  as f:      writer = csv.writer(f)      writer.writerow(["Address  Pair",  "Total  Packets",  "Total  Bytes",  "Avg  
Inter-Packet
 
Time
 
(s)",  "Protocols"])       for pair,  data in stats.items():          #  Calculate  Avg  Inter-packet  time          avg_diff = 0          if len(data['times'])  > 1:              diffs = [data['times'][i+1]  - data['times'][i]  for i in range(len(data['times'])-1)]              avg_diff = sum(diffs)  / len(diffs)                  writer.writerow([              f"{pair[0]} <->  {pair[1]}",              data['packets'],              data['bytes'],              f"{avg_diff:.6f}",              ",  ".join(data['protocols'])          ])   print("Conversation  analysis  saved  to  conversations.csv")
