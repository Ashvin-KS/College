#include<vector>

usingnamespace ns3;
NS_LOG_COMPONENT_DEFINE("CombinedMultiChatApp");
classChatServerApp:publicApplication{
public:
ChatServerApp():m_udpSocket(0),m_tcpSocket(0),m_udpPort(9000),
m_tcpPort(9001){}
private:
virtualvoidStartApplication(void){

// Setup UDP
m_udpSocket =Socket::CreateSocket(GetNode(),
UdpSocketFactory::GetTypeId());
m_udpSocket->Bind(InetSocketAddress(Ipv4Address::GetAny(),
m_udpPort));
m_udpSocket->SetRecvCallback(MakeCallback
(&ChatServerApp::HandleReadUdp,this));
// Setup TCP
m_tcpSocket =Socket::CreateSocket(GetNode(),
TcpSocketFactory::GetTypeId());
m_tcpSocket->Bind(InetSocketAddress(Ipv4Address::GetAny(),
m_tcpPort));
m_tcpSocket->Listen();
m_tcpSocket->SetAcceptCallback(
MakeCallback(&ChatServerApp::HandleRequestTcp,this),
MakeCallback(&ChatServerApp::HandleAcceptTcp,this)
);
}
boolHandleRequestTcp(Ptr<Socket> socket,const Address& from){
returntrue;
}
voidHandleAcceptTcp(Ptr<Socket> socket,const Address& from){
socket->SetRecvCallback(MakeCallback(&ChatServerApp::HandleReadTcp,
this));
// FIX: Store the accepted connection so it doesn't get deleted
m_tcpAcceptedSockets.push_back(socket);
}
voidProcessAndReply(Ptr<Socket> socket, Ptr<Packet> packet, Address
from, std::string protocol){
if(packet->GetSize()==0)return;
uint32_t size = packet->GetSize();
double timestamp =Simulator::Now().GetSeconds();
if(protocol =="TCP"){
socket->GetPeerName(from);
}
std::stringstream msg;
msg <<"ACK | Recv at: "<< timestamp <<"s | Size: "<< size <<"
bytes";
std::string responseString = msg.str();

std::cout <<"Server ["<< protocol <<" Processing]: Message from "
<<InetSocketAddress::ConvertFrom(from).GetIpv4()<<
std::endl;
Ptr<Packet> responsePkt =Create<Packet>
((uint8_t*)responseString.c_str(), responseString.length());
if(protocol =="UDP"){
socket->SendTo(responsePkt,0, from);
}else{
socket->Send(responsePkt);
}
}
voidHandleReadUdp(Ptr<Socket> socket){
Ptr<Packet> packet; Address from;
while((packet = socket->RecvFrom(from))){ProcessAndReply(socket,
packet, from,"UDP");}
}
voidHandleReadTcp(Ptr<Socket> socket){
Ptr<Packet> packet; Address from;
while((packet = socket->RecvFrom(from))){ProcessAndReply(socket,
packet, from,"TCP");}
}
Ptr<Socket> m_udpSocket;
Ptr<Socket> m_tcpSocket;
std::vector<Ptr<Socket>> m_tcpAcceptedSockets;// Keeps TCP sockets alive
uint16_t m_udpPort;
uint16_t m_tcpPort;
};
classChatClientApp:publicApplication{
public:
ChatClientApp():m_socket(0){}
voidSetup(Address address, Time interval, std::string clientName, TypeId
tid){
m_peer = address; m_interval = interval; m_name = clientName; m_tid =
tid;
}
private:
virtualvoidStartApplication(void){
m_socket =Socket::CreateSocket(GetNode(), m_tid);
m_socket->Connect(m_peer);
m_socket->SetRecvCallback(MakeCallback

(&ChatClientApp::HandleResponse,this));
Simulator::Schedule(Seconds(0.1),&ChatClientApp::SendChat,this);
}
voidSendChat(){
std::string msg ="Hello from "+ m_name;
Ptr<Packet> p =Create<Packet>((uint8_t*)msg.c_str(), msg.length
());
m_socket->Send(p);
Simulator::Schedule(m_interval,&ChatClientApp::SendChat,this);
}
voidHandleResponse(Ptr<Socket> socket){
Ptr<Packet> packet; Address from;
while((packet = socket->RecvFrom(from))){
if(packet->GetSize()==0)continue;
std::vector<uint8_t>buf(packet->GetSize());
packet->CopyData(buf.data(), packet->GetSize());
std::string data = std::string((char*)buf.data(), packet->GetSize
());
std::cout <<"["<< m_name <<"] Recv -> "<< data << std::endl;
}
}
Ptr<Socket> m_socket; Address m_peer; Time m_interval; std::string m_name;
TypeId m_tid;
};
